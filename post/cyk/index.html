/*
输入格式：
    第一行输入n, 表示生成式的个数, 输入要满足CNF.
    之后n行，每行代表一个生成式。用大写字母代表变量, 其中S表示起始变量,小写字母代表终结符，若有epsilon, 用0表示。(变量和终结符个数<=26)
    生成式如下 S->AB, S->AB|a, S->a|b, 中间没有多余字符。
    之后输入一个字符串，长度m <= 500.
复杂度：
    O(m^3*n)
*/
#include <bits/stdc++.h>
using namespace std;
#define mkp make_pair
const int N = 505;

int n, m;
bool f[N][N][27];
int from[N][N][27];
vector<int> who[27]; // 存可以生成终结符i的变量
pair<int, int> production[27]; // 存生成式
string w;
bool Sto0;

inline void readALine() {
    string x; getline(cin, x);
    char S = x[0];
    if (x[3] == '0') { // S->0
        Sto0 = true;
    } else { // S->AB...
        production[S-'A'] = mkp(x[3]-'A', x[4]-'A');
        for (int i = 0; i < x.length(); i ++)
			if (x[i] <= 'z' and x[i] >= 'a') 
            	who[x[i]-'a'].push_back(S-'A');
    }
}

inline void debug() {
    for (int i = 0; i < m; i ++)
        for (int j = i; j < m; j ++) {
            vector<int> a;
            for (int k = 0; k < 26; k ++)
                if (f[i][j][k]) a.push_back(k);
            if (a.size()) {
                printf("%d-%d : ", i+1, j+1);
                for (auto x: a) printf("%c ", x+'A');
                puts("");
            }
        }
}

inline void printAPath() {
    queue <pair< pair<int, int>, int> > q;
    q.push(mkp(mkp(0, m-1), 'S'-'A'));

    bool ok = 0;

    while(!q.empty()) {
        int l = q.front().first.first, r = q.front().first.second, p = q.front().second;
        q.pop();
        printf("%c", p+'A');
        if (l == r) {
            q.push(mkp(mkp(l, l), p));
        } else ok = 0;
        if (r == m - 1) {
            printf(" -> ");
            if (ok) break;
            ok = 1;
        }
        if (l == r) continue;
        int k = from[l][r][p];
        if (f[l][k][production[p].first] and f[k+1][r][production[p].second]) {
            q.push(mkp(mkp(l, k), production[p].first));
            q.push(mkp(mkp(k+1, r), production[p].second));
        }
    }
    cout << w << endl;
}

signed main() {
    freopen("test.txt", "r", stdin);
    cin >> n; getchar();
    for (int i = 1; i <= n; i ++) readALine();
    cin >> w;
    m = w.length();
    if (Sto0) { // 如果 S-> epsilon
        if (m == 1 and w[0] == '0') puts("Yes");
        else puts("No");
        return 0;
    }
    
    for (int i = 0; i < m; i ++) 
        for (auto x : who[w[i]-'a']) 
            f[i][i][x] = 1, from[i][i][x] = i;

    for (int l = 2; l <= m; l ++) {
        for (int i = 0 ; i < m; i ++) {
            int j = i + l - 1;
            if (j >= m) break;
            for (int k = i; k < j; k ++) {
                for (int p = 0; p < 26; p ++) {
                    int A = production[p].first, B = production[p].second;
                    if (A == 0 and B == 0) continue;
                    if (f[i][k][A] and f[k+1][j][B]) f[i][j][p] = 1, from[i][j][p] = k;
                }
            }
        }
    }

    debug();

    if (f[0][m-1]['S'-'A']) puts("Yes"), printAPath();
    else puts("No");
    return 0;
}
